def has_close_elements(numbers: List[float], threshold: float) -> bool:
    close_pairs = []
    for num in numbers:
        for i in range(len(numbers) - 1):
            if abs(num - numbers[i + 1]) <= threshold:
                close_pairs.append((num, numbers[i + 1]))
    return len(close_pairs) > 0
