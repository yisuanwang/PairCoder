def anti_shuffle(s):
    """
    Write a function that takes a string and returns an ordered version of it.
    Ordered version of string, is a string where all words (separated by space)
    are replaced by a new word where all the characters arranged in
    ascending order based on ascii value.
    Note: You should keep the order of words and blank spaces in the sentence.

    For example:
    anti_shuffle('Hi') returns 'Hi'
    anti_shuffle('hello') returns 'ehllo'
    anti_shuffle('Hello World!!!') returns 'Hello !!!Wdlor'
    """
    ordered_words = []

    # Split the input string by spaces
    words = s.split()

    # Iterate through each word
    for word in words:
        # Sort the characters in the word based on ascii value
        ordered_word = ''.join(sorted(word))
        # Add the ordered word to the list
        ordered_words.append(ordered_word)

    # Join the ordered words with spaces to recreate the original sentence
    ordered_sentence = ' '.join(ordered_words)

    return ordered_sentence
